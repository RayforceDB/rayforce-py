<pre><code class="language-python"><span class="code-comment"># Install Rayforce-Py</span>
<span class="code-keyword">pip</span> install rayforce-py

<span class="code-comment"># Basic usage</span>
<span class="code-keyword">import</span> rayforce <span class="code-keyword">as</span> rf

<span class="code-comment"># Create a table</span>
table = rf.table({
    <span class="code-string">'name'</span>: [<span class="code-string">'Alice'</span>, <span class="code-string">'Bob'</span>, <span class="code-string">'Charlie'</span>],

})

<span class="code-comment"># Query the table</span>
result = table.select(<span class="code-string">'name'</span>, <span class="code-string">'salary'</span>).where(table[<span class="code-string">'salary'</span>] > <span class="code-number">80000</span>)</code></pre>

result = table.select(

).group_by(<span class="code-string">'dept'</span>)

<span class="code-comment"># Join operations</span>

})</code></pre>

document.addEventListener('DOMContentLoaded', function() {
  // Intersection Observer for scroll animations
  const observerOptions = {
    threshold: 0.1,
    rootMargin: '0px 0px -50px 0px'
  };

  const observer = new IntersectionObserver(function(entries) {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        entry.target.classList.add('visible');
      }
    });
  }, observerOptions);

  // Observe all elements with scroll-fade-in class except logo
  document.querySelectorAll('.scroll-fade-in').forEach(el => {
    if (!el.matches('.hero-content > .scroll-fade-in:first-child')) {
      observer.observe(el);
    } else {
      el.classList.add('visible');
    }
  });
});
</script>
